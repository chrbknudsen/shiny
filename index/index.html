get_meta <- function(x) {
  res <- tryCatch({
    read_lines(paste0(x, "/meta.txt"))
  }, error = function(e) {
    "not found"
  })
  return(res)
}
get_logo <- function(x){
  sti <- paste0(x, "/logo.png")
  if(file.exists(paste0(sti))){
    return(sti)
  }else{
    return("not found")
  }
}
